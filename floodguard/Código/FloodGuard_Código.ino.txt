// TFT ILI9341 (SPI)
#define TFT_CS    10
#define TFT_DC     9
#define TFT_RST    8
#define TFT_MOSI  11
#define TFT_SCK   14
// MISO não usado (sem touch SPI aqui)

// HC-SR04
#define HCSR04_TRIG   12
#define HCSR04_ECHO   13

// Sensor de Chuva FR-04
#define RAIN_A0       34   // Analógico
#define RAIN_D0       35   // Digital

// DHT11 (embutido na placa de expansão)
#define DHT_PIN        3   // Verificar serigrafia

// Semáforo LED Open-Smart
#define LED_RED        4
#define LED_YELLOW     5
#define LED_GREEN      6

// Buzzer
#define BUZZER_PIN    15

// ─────────────────────────────────────
//  LIMITES DE ALERTA (cm)
//  Ajuste conforme seu reservatório/local
// ─────────────────────────────────────
#define DIST_MAXIMA     100  // distância = recipiente/rio VAZIO (cm)
#define DIST_ALERTA      50  // nível >= 50% → ALERTA
#define DIST_CRITICO     20  // nível >= 80% → CRÍTICO
// Quando dist medida for MENOR que limite → nível MAIS ALTO
// Ex: dist=20cm com máx=100cm → nível = 80%

// ─────────────────────────────────────
//  CORES ILI9341 (RGB565)
// ─────────────────────────────────────
#define C_BLACK     0x0000
#define C_WHITE     0xFFFF
#define C_BGDARK    0x0208   // #040a08 — verde muito escuro
#define C_PANEL     0x0410   // #080f10
#define C_BORDER    0x0821   // #0a1a10
#define C_GREEN     0x07E0   // verde puro
#define C_LGREEN    0x47E8   // verde claro
#define C_CYAN      0x07FF   // ciano
#define C_DCYAN     0x03EF   // ciano escuro
#define C_YELLOW    0xFFE0   // amarelo
#define C_ORANGE    0xFBE0   // laranja
#define C_RED       0xF800   // vermelho
#define C_DRED      0x8000   // vermelho escuro
#define C_GRAY      0x4208   // cinza escuro
#define C_DGRAY     0x2104   // cinza muito escuro
#define C_PURPLE    0x801F   // roxo

// ─────────────────────────────────────
//  OBJETOS GLOBAIS
// ─────────────────────────────────────
Adafruit_ILI9341 tft = Adafruit_ILI9341(TFT_CS, TFT_DC, TFT_MOSI, TFT_SCK, TFT_RST);
DHT dht(DHT_PIN, DHT_TYPE);
WiFiClient   wifiClient;
PubSubClient mqttClient(wifiClient);

// ─────────────────────────────────────
//  DADOS DOS SENSORES
// ─────────────────────────────────────
struct SensorData {
  float  distCm      = 0;
  int    nivelPct    = 0;
  int    rainA0      = 0;
  bool   rainDigital = false;
  float  tempC       = 0;
  float  humidity    = 0;
  int    rssi        = 0;
  bool   mqttOk      = false;
  uint32_t pubCount  = 0;
  uint32_t uptimeSec = 0;
};

SensorData data;

// ─────────────────────────────────────
//  ESTADO DO SISTEMA
// ─────────────────────────────────────
enum AlertLevel { NORMAL, ALERTA, CRITICO };
AlertLevel currentAlert = NORMAL;

enum Page { PAGE_HOME, PAGE_SENSORS, PAGE_ALERT, PAGE_MQTT };
Page currentPage = PAGE_HOME;

// ─────────────────────────────────────
//  HISTÓRICO (minigrafe)
// ─────────────────────────────────────
#define HISTORY_LEN 60
uint8_t history[HISTORY_LEN];
uint8_t histIdx = 0;
bool    histFull = false;

// ─────────────────────────────────────
//  TIMERS
// ─────────────────────────────────────
unsigned long tSensor    = 0;   // leitura sensores (500ms)
unsigned long tDisplay   = 0;   // atualização display (1s)
unsigned long tMQTT      = 0;   // publicação MQTT (5s)
unsigned long tUptime    = 0;   // contador uptime (1s)
unsigned long tBuzzer    = 0;   // controle buzzer
unsigned long tPageAuto  = 0;   // auto-page rotate (10s)
bool buzzerState = false;

// ─────────────────────────────────────
//  CONTROLE DE REDESENHO PARCIAL
// ─────────────────────────────────────
AlertLevel lastAlert    = NORMAL;
Page       lastPage     = PAGE_HOME;
int        lastNivel    = -1;
bool       forceRedraw  = true;

// ═══════════════════════════════════════════════════════════════
//  SETUP
// ═══════════════════════════════════════════════════════════════
void setup() {
  Serial.begin(115200);
  delay(300);
  Serial.println(F("\n[FLOOD·GUARD] Iniciando..."));

  // Pinos
  pinMode(HCSR04_TRIG,  OUTPUT);
  pinMode(HCSR04_ECHO,  INPUT);
  pinMode(RAIN_D0,      INPUT);
  pinMode(LED_RED,      OUTPUT);
  pinMode(LED_YELLOW,   OUTPUT);
  pinMode(LED_GREEN,    OUTPUT);
  pinMode(BUZZER_PIN,   OUTPUT);
  analogReadResolution(12);

  digitalWrite(LED_RED,    LOW);
  digitalWrite(LED_YELLOW, LOW);
  digitalWrite(LED_GREEN,  HIGH);  // Verde ligado na inicialização
  digitalWrite(BUZZER_PIN, LOW);

  // DHT
  dht.begin();

  // TFT
  tft.begin();
  tft.setRotation(1);          // Paisagem: 320×240
  tft.fillScreen(C_BLACK);
  Serial.println(F("[TFT] Iniciado 320x240"));

  // Tela de boot
  drawBootScreen();
  delay(1500);

  // WiFi
  connectWiFi();

  // MQTT
  mqttClient.setServer(MQTT_BROKER, MQTT_PORT);
  mqttClient.setKeepAlive(30);

  // Histórico zerado
  memset(history, 0, sizeof(history));

  // Força desenho completo
  forceRedraw = true;
  tft.fillScreen(C_BGDARK);

  Serial.println(F("[FLOOD·GUARD] Pronto!"));
}

// ═══════════════════════════════════════════════════════════════
//  LOOP
// ═══════════════════════════════════════════════════════════════
void loop() {
  unsigned long now = millis();

  // ── Sensores (500ms) ──
  if (now - tSensor >= 500) {
    tSensor = now;
    readSensors();
    updateAlertLevel();
    updateLEDs();
  }

  // ── Buzzer (modo crítico) ──
  if (currentAlert == CRITICO) {
    if (now - tBuzzer >= 600) {
      tBuzzer = now;
      buzzerState = !buzzerState;
      digitalWrite(BUZZER_PIN, buzzerState ? HIGH : LOW);
    }
  } else {
    digitalWrite(BUZZER_PIN, LOW);
    buzzerState = false;
  }

  // ── Uptime ──
  if (now - tUptime >= 1000) {
    tUptime = now;
    data.uptimeSec++;
  }

  // ── Display (1s) ──
  if (now - tDisplay >= 1000 || forceRedraw) {
    tDisplay = now;
    drawCurrentPage();
    forceRedraw = false;
  }

  // ── MQTT (5s) ──
  if (now - tMQTT >= 5000) {
    tMQTT = now;
    mqttLoop();
  }

  // ── MQTT keep-alive ──
  mqttClient.loop();
}

// ═══════════════════════════════════════════════════════════════
//  LEITURA DE SENSORES
// ═══════════════════════════════════════════════════════════════
void readSensors() {
  // HC-SR04
  data.distCm = readHCSR04();
  if (data.distCm <= 0 || data.distCm > DIST_MAXIMA) data.distCm = DIST_MAXIMA;
  data.nivelPct = constrain(
    (int)(((float)(DIST_MAXIMA - data.distCm) / DIST_MAXIMA) * 100.0f),
    0, 100
  );

  // FR-04 Chuva
  data.rainA0      = analogRead(RAIN_A0);
  data.rainDigital = (digitalRead(RAIN_D0) == HIGH);

  // DHT11
  float t = dht.readTemperature();
  float h = dht.readHumidity();
  if (!isnan(t)) data.tempC    = t;
  if (!isnan(h)) data.humidity = h;

  // WiFi RSSI
  data.rssi  = WiFi.RSSI();
  data.mqttOk = mqttClient.connected();

  // Histórico
  history[histIdx] = (uint8_t)data.nivelPct;
  histIdx = (histIdx + 1) % HISTORY_LEN;
  if (histIdx == 0) histFull = true;
}

// ── HC-SR04 ──
float readHCSR04() {
  digitalWrite(HCSR04_TRIG, LOW);
  delayMicroseconds(2);
  digitalWrite(HCSR04_TRIG, HIGH);
  delayMicroseconds(10);
  digitalWrite(HCSR04_TRIG, LOW);
  long dur = pulseIn(HCSR04_ECHO, HIGH, 25000); // timeout 25ms
  if (dur == 0) return -1;
  return dur * 0.0343f / 2.0f;
}

// ═══════════════════════════════════════════════════════════════
//  NÍVEL DE ALERTA
// ═══════════════════════════════════════════════════════════════
void updateAlertLevel() {
  AlertLevel prev = currentAlert;
  if (data.nivelPct >= 80)
    currentAlert = CRITICO;
  else if (data.nivelPct >= 50)
    currentAlert = ALERTA;
  else
    currentAlert = NORMAL;

  if (currentAlert != prev) {
    forceRedraw = true;
    Serial.printf("[ALERTA] Mudou: %d → %d\n", prev, currentAlert);
  }
}

// ═══════════════════════════════════════════════════════════════
//  LEDs SEMÁFORO
// ═══════════════════════════════════════════════════════════════
void updateLEDs() {
  digitalWrite(LED_RED,    currentAlert == CRITICO ? HIGH : LOW);
  digitalWrite(LED_YELLOW, currentAlert == ALERTA  ? HIGH : LOW);
  digitalWrite(LED_GREEN,  currentAlert == NORMAL  ? HIGH : LOW);
}

// ═══════════════════════════════════════════════════════════════
//  WIFI
// ═══════════════════════════════════════════════════════════════
void connectWiFi() {
  Serial.printf("[WiFi] Conectando a %s", WIFI_SSID);
  tft.fillRect(0, 140, 320, 20, C_BGDARK);
  tft.setTextColor(C_CYAN); tft.setTextSize(1);
  tft.setCursor(10, 145);
  tft.print("WiFi: ");
  tft.print(WIFI_SSID);

  WiFi.begin(WIFI_SSID, WIFI_PASSWORD);
  int tries = 0;
  while (WiFi.status() != WL_CONNECTED && tries < 30) {
    delay(500); Serial.print(".");
    tries++;
  }
  if (WiFi.status() == WL_CONNECTED) {
    Serial.printf("\n[WiFi] OK — IP: %s\n", WiFi.localIP().toString().c_str());
    tft.fillRect(0, 160, 320, 20, C_BGDARK);
    tft.setTextColor(C_GREEN); tft.setCursor(10, 165);
    tft.print("IP: "); tft.print(WiFi.localIP());
  } else {
    Serial.println(F("\n[WiFi] FALHOU — modo offline"));
    tft.setTextColor(C_YELLOW); tft.setCursor(10, 165);
    tft.print("Offline — sem WiFi");
  }
  delay(600);
}

// ═══════════════════════════════════════════════════════════════
//  MQTT
// ═══════════════════════════════════════════════════════════════
void mqttLoop() {
  if (WiFi.status() != WL_CONNECTED) return;
  if (!mqttClient.connected()) {
    Serial.print(F("[MQTT] Reconectando..."));
    if (mqttClient.connect(MQTT_CLIENT)) {
      Serial.println(F(" OK"));
    } else {
      Serial.printf(" FALHOU rc=%d\n", mqttClient.state());
      return;
    }
  }

  // Monta payload JSON
  StaticJsonDocument<256> doc;
  doc["nivel_pct"]  = data.nivelPct;
  doc["dist_cm"]    = data.distCm;
  doc["chuva"]      = data.rainDigital;
  doc["rain_a0"]    = data.rainA0;
  doc["temp_c"]     = data.tempC;
  doc["umidade"]    = data.humidity;
  doc["alerta"]     = alertName();
  doc["uptime"]     = data.uptimeSec;

  char buf[256];
  serializeJson(doc, buf);
  mqttClient.publish(MQTT_TOPIC, buf);
  data.pubCount++;
  data.mqttOk = true;

  Serial.printf("[MQTT] PUB #%lu → nivel=%d%% dist=%.1fcm\n",
    data.pubCount, data.nivelPct, data.distCm);
}

const char* alertName() {
  if (currentAlert == CRITICO) return "CRITICO";
  if (currentAlert == ALERTA)  return "ALERTA";
  return "NORMAL";
}

// ═══════════════════════════════════════════════════════════════
//  ROTEADOR DE PÁGINAS
// ═══════════════════════════════════════════════════════════════
void drawCurrentPage() {
  bool pageChanged = (currentPage != lastPage);
  bool alertChanged = (currentAlert != lastAlert);

  if (pageChanged || alertChanged) {
    tft.fillScreen(C_BGDARK);
    forceRedraw = true;
  }

  switch (currentPage) {
    case PAGE_HOME:    drawPageHome();    break;
    case PAGE_SENSORS: drawPageSensors(); break;
    case PAGE_ALERT:   drawPageAlert();   break;
    case PAGE_MQTT:    drawPageMQTT();    break;
  }

  lastPage  = currentPage;
  lastAlert = currentAlert;

  // Avança página automaticamente a cada 8s
  static unsigned long tAutoPage = 0;
  if (millis() - tAutoPage >= 8000) {
    tAutoPage = millis();
    currentPage = (Page)((currentPage + 1) % 4);
    tft.fillScreen(C_BGDARK);
  }
}

// Cor do alerta atual
uint16_t alertColor() {
  if (currentAlert == CRITICO) return C_RED;
  if (currentAlert == ALERTA)  return C_YELLOW;
  return C_GREEN;
}

// Topbar padrão (altura 20px)
void drawTopbar(const char* title, uint16_t titleColor) {
  tft.fillRect(0, 0, 320, 20, C_PANEL);
  tft.drawFastHLine(0, 20, 320, C_BORDER);

  // Título
  tft.setTextColor(titleColor);
  tft.setTextSize(1);
  tft.setCursor(6, 6);
  tft.print(title);

  // Status pill (direita)
  uint16_t pillColor = alertColor();
  char statusStr[10];
  if (currentAlert == CRITICO) strcpy(statusStr, "CRITICO");
  else if (currentAlert == ALERTA) strcpy(statusStr, "ALERTA");
  else strcpy(statusStr, "NORMAL");

  int sw = strlen(statusStr) * 6 + 8;
  int sx = 316 - sw;
  tft.drawRect(sx, 3, sw, 14, pillColor);
  tft.setTextColor(pillColor);
  tft.setCursor(sx + 4, 7);
  tft.print(statusStr);

  // Relógio (centro)
  // Uptime como clock aproximado
  char clk[10];
  uint32_t h = data.uptimeSec / 3600;
  uint32_t m = (data.uptimeSec % 3600) / 60;
  uint32_t s = data.uptimeSec % 60;
  snprintf(clk, sizeof(clk), "%02lu:%02lu:%02lu", h, m, s);
  tft.setTextColor(C_GRAY);
  tft.setCursor(134, 6);
  tft.print(clk);
}

// Bottombar navegação (altura 22px, y começa em 218)
void drawBottombar(Page activePage) {
  tft.fillRect(0, 218, 320, 22, C_PANEL);
  tft.drawFastHLine(0, 218, 320, C_BORDER);

  const char* labels[4] = { "HOME", "SENSOR", "ALERTA", "MQTT" };
  int w = 80;
  for (int i = 0; i < 4; i++) {
    int x = i * w;
    if (i == (int)activePage) {
      tft.fillRect(x, 219, w - 1, 21, C_BORDER);
      tft.setTextColor(alertColor());
    } else {
      tft.setTextColor(C_DGRAY);
    }
    // Centraliza texto
    int tx = x + (w - strlen(labels[i]) * 6) / 2;
    tft.setCursor(tx, 225);
    tft.setTextSize(1);
    tft.print(labels[i]);
    // Separador vertical
    if (i < 3) tft.drawFastVLine(x + w - 1, 218, 22, C_BORDER);
  }
  // Indicador ativo (linha colorida no topo da bottombar)
  tft.drawFastHLine((int)activePage * w, 218, w - 1, alertColor());
}

// Barra de progresso horizontal
void drawHBar(int x, int y, int w, int h, int pct, uint16_t fillColor, uint16_t bgColor) {
  tft.fillRect(x, y, w, h, bgColor);
  tft.drawRect(x, y, w, h, C_BORDER);
  if (pct > 0) {
    int fw = constrain((int)(pct / 100.0f * (w - 2)), 0, w - 2);
    tft.fillRect(x + 1, y + 1, fw, h - 2, fillColor);
  }
}

// Barra vertical (tanque d'água)
void drawVBar(int x, int y, int w, int h, int pct, uint16_t fillColor) {
  tft.fillRect(x, y, w, h, C_DGRAY);
  tft.drawRect(x, y, w, h, C_GRAY);
  // Marcas de risco
  int yWarn = y + h - (int)(h * 0.50f);
  int yCrit = y + h - (int)(h * 0.80f);
  tft.drawFastHLine(x - 2, yWarn, w + 4, C_YELLOW);
  tft.drawFastHLine(x - 2, yCrit, w + 4, C_RED);
  if (pct > 0) {
    int fh = constrain((int)(pct / 100.0f * (h - 2)), 0, h - 2);
    tft.fillRect(x + 1, y + h - 1 - fh, w - 2, fh, fillColor);
  }
}

// Texto grande (número)
void drawBigNumber(int x, int y, float val, int decimals, const char* unit, uint16_t color) {
  tft.setTextColor(color);
  tft.setTextSize(3);
  char buf[12];
  if (decimals == 0) snprintf(buf, sizeof(buf), "%d", (int)val);
  else snprintf(buf, sizeof(buf), "%.1f", val);
  tft.setCursor(x, y);
  tft.print(buf);
  tft.setTextSize(1);
  tft.setTextColor(C_GRAY);
  tft.setCursor(x + strlen(buf) * 18 + 2, y + 14);
  tft.print(unit);
}

// Etiqueta + valor em linha
void drawLabelValue(int x, int y, const char* label, const char* value, uint16_t valColor) {
  tft.setTextColor(C_DGRAY); tft.setTextSize(1);
  tft.setCursor(x, y);
  tft.print(label);
  tft.setTextColor(valColor);
  tft.setCursor(x + strlen(label) * 6 + 4, y);
  tft.print(value);
}

// Mini-gráfico de histórico (linha simples)
void drawMiniChart(int x, int y, int w, int h) {
  tft.fillRect(x, y, w, h, 0x0208);
  tft.drawRect(x, y, w, h, C_BORDER);

  // Linhas de zona
  int yw = y + h - (int)(h * 0.50f);
  int yc = y + h - (int)(h * 0.80f);
  tft.drawFastHLine(x + 1, yw, w - 2, 0x4400); // amarelo escuro
  tft.drawFastHLine(x + 1, yc, w - 2, 0x4000); // vermelho escuro

  // Conta pontos disponíveis
  int total = histFull ? HISTORY_LEN : histIdx;
  if (total < 2) return;

  float step = (float)(w - 2) / (float)(total - 1);
  int prevX = -1, prevY = -1;
  for (int i = 0; i < total; i++) {
    int hi = (histIdx - total + i + HISTORY_LEN) % HISTORY_LEN;
    int px = x + 1 + (int)(i * step);
    int py = y + h - 1 - (int)(history[hi] / 100.0f * (h - 2));
    if (prevX >= 0) {
      tft.drawLine(prevX, prevY, px, py, C_DCYAN);
    }
    prevX = px; prevY = py;
  }
  // Ponto atual
  tft.fillCircle(prevX, prevY, 2, C_CYAN);
}

// ═══════════════════════════════════════════════════════════════
//  PÁGINA 1 — HOME
// ═══════════════════════════════════════════════════════════════
void drawPageHome() {
  drawTopbar("FLOOD GUARD", C_LGREEN);

  // ── Coluna esquerda: tanque + nivel ──
  int tankX = 8, tankY = 28, tankW = 18, tankH = 155;
  uint16_t waterColor = alertColor();

  drawVBar(tankX, tankY, tankW, tankH, data.nivelPct, waterColor);

  // % nível

  char pctStr[6]; snprintf(pctStr, sizeof(pctStr), "%d%%", data.nivelPct);
  tft.setCursor(34, 62);
  tft.print(pctStr);
  tft.setTextColor(C_DGRAY); tft.setTextSize(1);
  tft.setCursor(34, 82);
  tft.print("NIVEL RELATIVO");

  // Barra de risco
  tft.setTextColor(C_DGRAY); tft.setCursor(34, 95); tft.print("RISCO:");

  char s1[12]; snprintf(s1, sizeof(s1), "%.0fcm", data.distCm);

  int p2 = (int)(data.rainA0 / 4095.0f * 100);

  drawSensorRow("[R]", "FR-04  CHUVA ", p2, C_CYAN, s2);

  // DHT11 Temp
  int p3 = (int)(data.tempC / 50.0f * 100);
  char s3[8]; snprintf(s3, sizeof(s3), "%.0fC", data.tempC);
  drawSensorRow("[T]", "DHT11  TEMP  ", p3, C_ORANGE, s3);

  // DHT11 Umidade
  int p4 = (int)data.humidity;
  char s4[6]; snprintf(s4, sizeof(s4), "%d%%", p4);

  // WiFi RSSI
  int rssiPct = constrain((int)((data.rssi + 100) / 60.0f * 100), 0, 100);

  tft.setTextColor(C_DGRAY); tft.setCursor(6, y + 6); tft.print("HISTORICO NIVEL (60s)");
  drawMiniChart(6, y + 16, 308, 44);

  drawBottombar(PAGE_SENSORS);

void drawPageAlert() {
  drawTopbar("ALERTA", alertColor());

  // ── Semáforo (coluna esquerda, 80px) ──
  tft.drawFastVLine(80, 22, 193, C_BORDER);

  // Círculos dos LEDs
  int cx = 40;
  // Vermelho
  bool rOn = (currentAlert == CRITICO);
  tft.fillCircle(cx, 65,  20, rOn ? C_RED   : C_DGRAY);
  tft.drawCircle(cx, 65,  21, C_DRED);
  // Amarelo
  bool yOn = (currentAlert == ALERTA);
  tft.fillCircle(cx, 120, 20, yOn ? C_YELLOW : C_DGRAY);
  tft.drawCircle(cx, 120, 21, 0x4400);
  // Verde
  bool gOn = (currentAlert == NORMAL);
  tft.fillCircle(cx, 175, 20, gOn ? C_GREEN  : C_DGRAY);
  tft.drawCircle(cx, 175, 21, 0x0440);

  // Labels
  tft.setTextSize(1); tft.setCursor(26, 90);
  tft.setTextColor(rOn ? C_RED    : C_DGRAY); tft.print("CRIT");
  tft.setCursor(21, 145);
  tft.setTextColor(yOn ? C_YELLOW : C_DGRAY); tft.print("ALERT");
  tft.setCursor(25, 200);
  tft.setTextColor(gOn ? C_GREEN  : C_DGRAY); tft.print("NORM");

  // ── Painel direito ──
  int rx = 88;

  // Status grande
  tft.setTextColor(alertColor()); tft.setTextSize(3);
  tft.setCursor(rx, 30);
  tft.print(alertName());

  // Linha separadora
  tft.drawFastHLine(rx, 62, 225, C_BORDER);

  // Descrição
  tft.setTextSize(1); tft.setTextColor(C_GRAY);
  tft.setCursor(rx, 68);
  if (currentAlert == CRITICO) {
    tft.setTextColor(C_RED);
    tft.print("RISCO DE ENCHENTE!");
    tft.setCursor(rx, 80); tft.print("Acionar equipe de emergencia");
    tft.setCursor(rx, 92); tft.print("Emitir alerta a populacao");
  } else if (currentAlert == ALERTA) {
    tft.setTextColor(C_YELLOW);
    tft.print("Nivel d'agua em elevacao.");
    tft.setCursor(rx, 80); tft.print("Monitoramento intensificado.");
    tft.setCursor(rx, 92); tft.print("Equipes em standby.");
  } else {
    tft.setTextColor(C_GREEN);
    tft.print("Sistema operando normalmente.");
    tft.setCursor(rx, 80); tft.print("Monitoramento padrao ativo.");
    tft.setCursor(rx, 92); tft.print("Sem risco detectado.");
  }

  // Grade de dados
  tft.drawFastHLine(rx, 108, 225, C_BORDER);

  auto drawInfoCell = [&](int x, int y, int w, int h,
                          const char* k, const char* v, uint16_t vc) {
    tft.fillRect(x, y, w, h, C_PANEL);
    tft.drawRect(x, y, w, h, C_BORDER);
    tft.setTextColor(C_DGRAY); tft.setTextSize(1);
    tft.setCursor(x + 4, y + 4); tft.print(k);
    tft.setTextColor(vc); tft.setTextSize(2);
    tft.setCursor(x + 4, y + 16); tft.print(v);
  };

  char vNivel[6],  vDist[8], vChuva[4], vTemp[6];
  snprintf(vNivel, sizeof(vNivel), "%d%%",  data.nivelPct);
  snprintf(vDist,  sizeof(vDist),  "%.0fcm", data.distCm);
  snprintf(vChuva, sizeof(vChuva), "%s", data.rainDigital ? "SIM" : "NAO");
  snprintf(vTemp,  sizeof(vTemp),  "%.0fC",  data.tempC);

  drawInfoCell(rx,       112, 107, 44, "NIVEL",  vNivel, alertColor());
  drawInfoCell(rx + 113, 112, 112, 44, "DIST",   vDist,  C_CYAN);
  drawInfoCell(rx,       160, 107, 44, "CHUVA",  vChuva, data.rainDigital ? C_CYAN : C_DGRAY);
  drawInfoCell(rx + 113, 160, 112, 44, "TEMP",   vTemp,  C_ORANGE);

  // Buzzer ativo?
  if (currentAlert == CRITICO) {
    tft.setTextColor(C_RED); tft.setTextSize(1);
    tft.setCursor(rx, 210); tft.print(">>> BUZZER ATIVO <<<");
  }

  drawBottombar(PAGE_ALERT);
}

// ═══════════════════════════════════════════════════════════════
//  PÁGINA 4 — MQTT / SISTEMA
// ═══════════════════════════════════════════════════════════════
void drawPageMQTT() {
  drawTopbar("MQTT  SISTEMA", C_PURPLE);

  int y = 28, x = 8;

  auto drawSysRow = [&](const char* k, const char* v, uint16_t vc, int barPct = -1) {
    tft.fillRect(0, y, 320, 22, C_PANEL);
    tft.drawFastHLine(0, y, 320, C_BORDER);
    tft.setTextColor(C_DGRAY); tft.setTextSize(1);
    tft.setCursor(x, y + 7); tft.print(k);
    if (barPct >= 0) {
      drawHBar(x + 68, y + 8, 160, 5, barPct, C_PURPLE, C_DGRAY);
    }
    tft.setTextColor(vc); tft.setTextSize(1);
    int vx = barPct >= 0 ? (x + 234) : (x + 72);
    tft.setCursor(vx, y + 7); tft.print(v);
    y += 23;
  };

  // Status WiFi/MQTT
  drawSysRow("WiFi STATUS", WiFi.status() == WL_CONNECTED ? "CONECTADO" : "OFFLINE",
             WiFi.status() == WL_CONNECTED ? C_GREEN : C_RED);

  char ipStr[20];
  WiFi.localIP().toString().toCharArray(ipStr, sizeof(ipStr));
  drawSysRow("IP LOCAL   ", ipStr, C_GRAY);

  char rssiStr[12]; snprintf(rssiStr, sizeof(rssiStr), "%d dBm", data.rssi);
  int rssiPct = constrain((int)((data.rssi + 100) / 60.0f * 100), 0, 100);
  drawSysRow("WiFi RSSI  ", rssiStr, C_LGREEN, rssiPct);

  drawSysRow("MQTT BROKER", "hivemq.com", C_GRAY);
  drawSysRow("TOPICO     ", MQTT_TOPIC,   C_GRAY);

  char qosStr[4] = "1"; drawSysRow("QoS        ", qosStr, C_PURPLE);

  char pubStr[12]; snprintf(pubStr, sizeof(pubStr), "%lu", data.pubCount);
  int pubPct = constrain((int)(data.pubCount / 200.0f * 100), 0, 100);
  drawSysRow("PUBLICACOES", pubStr, C_PURPLE, pubPct);

  // Uptime
  uint32_t us = data.uptimeSec;
  char upStr[12];
  snprintf(upStr, sizeof(upStr), "%02lu:%02lu:%02lu", us/3600, (us%3600)/60, us%60);
  drawSysRow("UPTIME     ", upStr, C_GRAY);

  // Última leitura
  tft.fillRect(0, y, 320, 8, C_BGDARK);
  tft.drawFastHLine(0, y, 320, C_BORDER);
  y += 4;

  tft.setTextColor(C_DGRAY); tft.setCursor(x, y + 4); tft.print("ULTIMA LEITURA:");
  y += 14;
  char logStr[64];
  snprintf(logStr, sizeof(logStr), "nivel=%d%% dist=%.0fcm r=%d t=%.0f h=%.0f",
    data.nivelPct, data.distCm,
    data.rainDigital ? 1 : 0,
    data.tempC, data.humidity);
  tft.setTextColor(C_PURPLE); tft.setCursor(x, y + 4); tft.print(logStr);

  drawBottombar(PAGE_MQTT);
}
